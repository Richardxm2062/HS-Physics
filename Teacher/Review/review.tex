\section{匀变速直线运动}

\subsection{运动学公式}
$$
    \begin{cases}
        V_{t} = v_{0} + at                                        \\

        x = \frac{(v_{0} + v_{t})}{2} t = \overline{v} t          \\

        x = V_{0} + \frac{1}{2}at^{2} = v_{t} - \frac{1}{2}at_{2} \\

        v_{t}^{2} - v_{0}^{2} = 2ax
    \end{cases}
$$

总结点

\begin{itemize}
    \item 区分 \, 前秒内 \, 与 \, 第秒内 \,
    \item 刹车问题需要先求解制动时间或制动距离
    \item 自由落体运动的初速度为$0$,\textbf{仅}受到重力
    \item 竖直上抛运动的初速度为$v_{0}$向上,\textbf{仅}受到重力,同一高度的速率大小一样
    \item 竖直上抛运动时常要分析多解性(向上段或向下段)
\end{itemize}

\vspace{2em}

\subsection{中间时刻/平均速度}
中间时刻速度$v_{\frac{t}{2}}$与平均速度$\overline{v}$是同一个值
$$
    v_{\frac{t}{2}} = v_{0} + \frac{at}{2} = \frac{v_{0}}{2} +  (\frac{v_{0}}{2} + \frac{at}{2})   = \frac{v_{0}+v_{t}}{2} = \overline{v}
$$

中间位置速度

\begin{numcases}{}
    \label{1} 2 a\frac{x}{2} = v_{\frac{x}{2}}^{2} - v_{0}^{2}  \\
    \label{2} 2 a\frac{x}{2} = v_{t}^{2} - v_{\frac{x}{2}}^{2}
\end{numcases}

由方程$(1) - (2)$ 得到 $ v_{\frac{x}{2}} = \sqrt{\frac{v_{0}^{2} + v_{t}^{2}}{2}} $

\subsection{纸带加速度问题}
纸带的特点,每个计时点的时间间隔相同均为$T$,且$x_{n}$规定的是第$n$个时间间隔内的位移,并非到起点的距离

\begin{corollary*}
    相邻位移之间的差为$aT^{2}$,等时位移比例式为$x_{1}:x_{2}:x_{3} : \dots : x_{n} = 1:3:5: \dots :2n-1  $
\end{corollary*}
\begin{proof}
    \begin{align*}
        x_{n} = \frac{1}{2}a (nT)^{2} -  \frac{1}{2}a [(n-1)T]^{2} & = aT^{2} (\frac{2n-1}{2})                                                  \\
        x_{n-1}                                                    & = aT^{2} (\frac{2n-3}{2})                                                  \\
        x_{n} - x_{n-1}                                            & = aT^{2}                                                                   \\
        \text{一般性的\quad} x_{i} - x_{j}                             & = (i-j) \, aT^{2} \hspace{1em} (i = \sum{\text{指标}} > j = \sum{\text{指标}})
    \end{align*}
\end{proof}

通常打点计时器的频率为$50 \, Hz$,即$0.02 \,s $秒间隔打下计时点,点数太过密集通常取$6$个计时点(间隔$5$个间隙)作为
计数点,$T = 5 \cross 0.02 \,s = 0.1 \, s$

\vspace{2em}

\begin{corollary*}
    等位移比例式子($1m$,$2m$,$3m \dots$)    \\
    前$1m,2m,3m \dots n \, m$所用时间比为$1:\sqrt{2}:\sqrt{3}:\dots:\sqrt{n}$,若是第$i\,m$内则向前减一个就行
    \begin{proof}
        \begin{align*}
            1 & = \frac{1}{2}a t_{1}^{2} \lra t_{1} =\sqrt{\frac{2}{a}} \vdot \sqrt{1} \\
            2 & = \frac{1}{2}a t_{2}^{2} \lra t_{2} =\sqrt{\frac{2}{a}} \vdot \sqrt{2} \\
            3 & = \frac{1}{2}a t_{3}^{2} \lra t_{3} =\sqrt{\frac{2}{a}} \vdot \sqrt{3} \\
            n & = \frac{1}{2}a t_{n}^{2} \lra t_{n} =\sqrt{\frac{2}{a}} \vdot \sqrt{n} \\
        \end{align*}
    \end{proof}
\end{corollary*}

\section{机械振动}
\subsection{简谐振动}
\begin{itemize}
    \item 定义: 具有平衡位置,回复力形如$F_{\text{回}} = -kx$(来自合外力或其分力)
    \item 振子方程: $\sin{(\omega t + \varphi)}$
    \item 同侧法: 质点振动速度方向$v_{f}$与波传播方向$u$在正弦函数线的同一侧
    \item 摆周期: $T = 2\pi \sqrt{\frac{L}{g}}$
    \item 受迫振动:在周期性外力的持续作用下而进行的振动称为\textbf{受迫振动}

          \hspace{4.7em}振动稳定后其\textbf{频率}等于外力驱动频率
    \item 等效绳长与等效加速度问题:
          \begin{itemize}
              \item 等效绳长: 确定为简谐振动,通过几何关系确定摆心
              \item 等效加速度: 主要区别电场摆和电梯摆,后者需要变换参考系(非惯性力)
          \end{itemize}
    \item  造成波的多解性的三大原因:
          \begin{itemize}
              \item \textbf{波的周期性:}\hspace{1em}
                    $\begin{cases}
                            \text{时间周期性:时间间隔}\triangle t \text{与周期} T \text{的关系不明确} \\
                            \text{空间周期性:波传播距离}\triangle x \text{与波长} \lambda \text{的关系不明确}
                        \end{cases}$

              \item \textbf{波的双向性:}\hspace{1em}
                    $\begin{cases}
                            \text{传播方向双向性:波的传播方向不确定} \\
                            \text{振动方向双向性:质点振动方向不确定}
                        \end{cases}$

              \item \textbf{波形隐含性:}\hspace{1em}
                    $\begin{cases}
                            \text{在波动问题中,有时只给出几个特殊点} \\
                            \text{(大多是两个特殊的点)的运动状态,其余信息均处于隐含状态}
                        \end{cases}$
          \end{itemize}
\end{itemize}

\section{热学}
\subsection{分子动理论}
\subsubsection{物体是由大量分子构成的}
\begin{enumerate}
    \item 分子的大小
          \begin{enumerate}[label = (\arabic*{})]
              \item 分子直径数量级: $10^{-10} \, m$
              \item 分子质量数量级: $10^{-26} \, kg$
              \item 测量分子直径的方法: 油膜法
          \end{enumerate}

    \item 两种分子模型
          \begin{enumerate}[label = (\arabic*{})]
              \item 球体模型: 认为分子是一个个紧挨着的球体(适用对象: 固体,液体;体积:$V = \frac{4\pi}{3} R^{3}$)
              \item 立方体模型: 认为分子是一个个紧挨着的立方体(适用对象: 气体;体积:$V = d^{3}$)
          \end{enumerate}

          \vspace{-1em}
          \hspace{1em}\begin{adjustbox}{minipage=0.86\linewidth, bgcolor=gray!20, padding=1em}
              \small
              固液相使用球体模型更为准确,气相使用立方体模型更为准确,这里的立方体空间是\textbf{气体分子}的活动空间
          \end{adjustbox}
          \vspace{-1em}

          \vspace{2em}

    \item 油膜法测量分子的直径
          \begin{itemize}
              \item 思路: 滴一滴油酸(已知体积$V$)摊开在水面上,近似看成单分子层(厚度为$d$)

                    \hspace{2.7em}此滴油酸体积可求$V = S \vdot d $,测量面积$S$即可计算厚度$d$
              \item 模型: 认为油酸分子为球体模型
              \item 要点:
                    \begin{itemize}
                        \item $1ml$油酸加酒精配制成$500ml$油酸酒精溶液(酒精易挥发特点)$\hspace{1em}\lra$稀释
                        \item 取$100$滴油酸酒精溶液测得体积为$1ml$$\hspace{1em}\lra$控制体积
                        \item \textbf{先}对水撒痱子粉后滴液$\hspace{1em}\lra$解决油膜与水的界面不明显问题
                        \item 往水面上注射一滴油酸酒精溶液$\hspace{1em}\lra$尽可能少的油酸分子
                        \item 等待酒精挥发稳定后使用玻璃板盖进行描边$\hspace{1em}\lra$勾勒面积轮廓
                        \item 使用\textbf{格子法}来计算不规则面积(不足半格取$0$,超过半格取$1$)
                        \item 一滴$\frac{1}{100} \, ml$油酸酒精溶液(油酸浓度为$\frac{1}{500}$)
                        \item []
                              其油酸体积为$\frac{1}{100} \, ml \vdot \frac{1}{500} \, \vdot {10}^{-6}= 2 \times 10 ^{-11} \, m^{3}$
                    \end{itemize}
          \end{itemize}

          \subsubsection{微观量的估算}
          \begin{enumerate}[label = (\arabic*{})]
              \item 微观层面: 单个分子质量$m_{0}$ \quad 单个分子所占\textbf{空间体积}$V_{0}$ \quad 单个分子体积$V_{\text{分子}}$
              \item 化学层面: 物质的量$n \, (mol)$ \quad 阿伏伽德罗$N_{A} = 6.02 \cross 10^{23}$

                    \hspace{4.8em}摩尔质量$M_{mol} \, (g \slash mol)$ \quad 摩尔体积$V_{mol} \, (L \slash mol)$

              \item 宏观层面: 质量$M$,体积$V$,数量$N$
              \item[] 物理量之间的关系(从实际的物理含义去理解记忆):

                    \begin{enumerate}
                        \item 微观-化学 \quad $ N_{A} = \frac{M_{mol}}{m_{0}} = \frac{V_{mol}}{V_{0}}$

                        \item 微观-宏观 \quad $ N = \frac{M}{m_{0}} = \frac{V}{V_{0}}$


                        \item 化学-宏观 \quad $n = \frac{N}{N_{A}} = \frac{M}{M_{mol}} = \frac{V}{V_{mol}}$
                    \end{enumerate}
                    $$
                        \hspace{0.2em} \text{摩尔体积适用于固液相,但是计算出来的并不能视作单分子体积,而是单分子占用空间}
                    $$
                    $$
                        \hspace{-2em}\text{摩尔体积不适用于气体.标况下为}22.4L\text{,除以}N_{A}\text{会导致每个分子占用空间一样大}
                    $$

                    \vspace{-1em}
                    \hspace{-1em}
                    \begin{adjustbox}{minipage=0.36\linewidth, bgcolor=gray!20, padding=1em}
                        \small % 将字号变小为 small
                        标况: $0^{\circ}C$,一个标准大气压(约$101kPa$)
                    \end{adjustbox}

              \item 密度: 区分 \, 气体密度$\rho_{\text{气体}}$ \, 与 \, 分子密度$\rho_{\text{分子}}$

                    \hspace{2.8em}利用气体密度所求体积为 \,分子所占空间体积$V_{0}$

                    $$
                        \rho_{gas} = \frac{M}{V} = \frac{m_{0}}{V_{0}}    \quad \quad \quad   \rho_{\text{分子}} = \frac{m_{0}}{V_{\text{分子}}}
                    $$
          \end{enumerate}
\end{enumerate}

\subsubsection{扩散现象和布朗运动}
一切物质的分子都在不停地做无规则的(热)运动

\begin{enumerate}
    \item 扩散现象
          \begin{enumerate}[label = (\arabic*)]
              \item 定义: 相互接触的物质能够彼此进入对方

                    \hspace{2.7em}此现象并不是宏观受力的作用下发生的,且各个状态下都会存在扩散现象

                    \hspace{2.7em}微观作用力为分子间作用力,且热运动的动能表现为净物质的迁移
              \item 补充说明: 浓度差(浓度梯度力为宏观表象,且并非唯一因素。仅高中简化理解)

                    \hspace{4.7em}允许同一物质接触发生扩散(即使无浓度差也可以发生自扩散)
              \item \textbf{直接}反映了分子的无规则运动
              \item 扩散强度: 正相关于温度,气体$>$液体$>$固体
          \end{enumerate}
    \item 布朗运动
          \begin{enumerate}[label = (\arabic*)]
              \item 定义: 悬浮在液体或气体中的微粒(宏观层面)的无规则运动
              \item 实验背景: 布朗看水中的花粉(显微镜)
              \item 观察结论: 微粒越小或温度越高,布朗运动越强烈
              \item 原因: 水分子对布朗微粒撞击的不平衡(不均匀)
              \item 布朗运动\textbf{不是}分子运动,\textbf{间接}反映了分子的无规则运动
          \end{enumerate}
    \item 扩散与布朗运动的混淆点
          \begin{enumerate}[label = (\arabic*)]
              \item 扩散现象: 微观力作用(分子),宏观现象肉眼可见
              \item 布朗运动: 宏观力作用(撞击),宏观现象光学显微镜观察可见($10^{-6}$)
              \item 两者都是微观层面的分子无规则运动所形成的宏观现象
          \end{enumerate}
\end{enumerate}

\subsubsection{分子之间的作用力}
\begin{itemize}
    \item 现象
          \begin{enumerate}[label = (\arabic*)]
              \item 分子虽然有空隙,大量分子聚集形成固体或液体说明了分子之间存在\textbf{引力}
              \item 用力压缩物体,物体内会产生反抗压缩的弹力,说明了分子之间存在斥力
          \end{enumerate}
    \item 研究表明
          \begin{enumerate}[label = (\arabic*)]
              \item 分子之间引力和斥力\textbf{同时存在}
              \item 引力与斥力都随着分子间距离增大而减小
              \item 斥力的变化比引力的变化更加明显
          \end{enumerate}
    \item 结论
          \begin{enumerate}[label = (\arabic*)]
              \item 分子距离较近时$<r_{0}$体现为 \, \textbf{斥力}
              \item 分子距离较远时$>r_{0}$体现为 \, \textbf{引力}
              \item 分子距离为$=r_{0}$合力为$0$
              \item 当分子间距离$>10 \, r_{0}$,引力、斥力均忽略不计
          \end{enumerate}

\end{itemize}

\subsubsection{气体}
\begin{enumerate}
    \item 气体压强的微观解释
          \begin{enumerate}[label = (\arabic*)]
              \item 气压: 气体分子对物质界面不断的撞击

              \item 推导(截面积法):
                    \begin{align*}
                        F \Delta t                      & = N \cdot 2mv \quad (\text{碰撞后原速反弹})                                                      \\
                        N                               & = \frac{1}{2} n \Delta V = \frac{1}{2} n (v \Delta t \cdot S) \quad (\text{仅一半粒子速度为碰撞方向}) \\
                        \Longrightarrow\quad F \Delta t & = \frac{1}{2} n v S \Delta t \cdot 2mv                                                    \\
                        \Longrightarrow\quad F          & = nmv^2 S                                                                                 \\
                        \Longrightarrow\quad P          & = \frac{F}{S} = nmv^2
                    \end{align*}

              \item 影响压强的因素:
                    $n$(单位体积分子数,与体积$V$有关)

                    \hspace{7.5em} $v$(分子热运动速度,与温度$T$有关)

                    \vspace{-1em}
                    \hspace{-0.9em}
                    \begin{adjustbox}{minipage=0.65\linewidth, bgcolor=gray!20, padding=1em}
                        \small
                        $\Phi = \frac{N}{S \Delta t}$ (分子通量)表示单位面积、单位时间的撞击次数\\
                        与分子数密$n$的关系为 $\displaystyle \Phi = \frac{1}{2}n\bar{v}$
                    \end{adjustbox}
                    \vspace{-1em}
          \end{enumerate}
\end{enumerate}

\subsubsection{气体三大实验定律}
\begin{enumerate}[label = \arabic*]
    \item 气体的状态参量
          \begin{enumerate}[label = (\arabic*)]
              \item 参量: $P \, V \, T$
              \item 关系: $\frac{PV}{T} = C$
              \item 适用条件: 同一气体且质量(物质的量)不变(当漏气时$C$下降)
          \end{enumerate}
    \item 玻意耳定律(人名要记忆)---$T$不变
          \begin{itemize}
              \item 概念: 温度一定时,一定质量的气体,$P,V$成反比$\lra PV = C$
              \item 记忆方式: \textbf{耳}的右边有类似$T$的存在
          \end{itemize}
    \item 查理定律(人名要记)---$V$不变
          \begin{itemize}
              \item 概念: 体积一定时,一定质量的气体,$P,T$成正比$\lra \frac{P}{T} = C$
              \item 记忆方式: \textbf{查}中间有个倒着的$V$
          \end{itemize}
    \item 盖-吕萨克定律(人名要记)---$P$不变
          \begin{itemize}
              \item 概念: 压强一定时,一定质量的气体,$V,T$成正比$\lra \frac{V}{T} = C$
              \item 记忆方式: \textbf{萨}左下角有个类似$P$的存在
          \end{itemize}
\end{enumerate}

\subsubsection{方法论}

\begin{itemize}
    \item 瞬态变化量的计算(液柱移动问题)

          $ \frac{P}{T} = \frac{\triangle P}{\triangle T} \quad or \quad \frac{V}{T} = \frac{\triangle V}{\triangle T}  \quad $(过原点的函数才能这样写)

          等式只用做计算$\triangle P \,or\, \triangle T \, or \, \triangle V$瞬间变化程度,以此来判断液面移动方向

          并不代表稳态后的$\triangle P \,or\, \triangle T \,or\, \triangle V$

          若在瞬态过程中,有同时有多个量导致液面的上升或下降

          可以直接考虑末态方程,从结果上判断某些量的确定增减性,在进行判断液面

    \item 充放气问题-物质的量的变化
          \begin{enumerate}
              \item 克拉伯龙方程: $PV = n R T \quad $($n$:物质的量;$R$:气体常数)
              \item 待充$or$待抽气体: $P\triangle V = \triangle n RT \quad $($n,V,m$具有等价性)
              \item 核心: \textbf{物质的量}守恒
              \item 充气: 待充气体的压强$P$通常会给出
                    $$
                        P_{1}V_{1} + n P\triangle{V} = P_{2}V_{2}
                    $$
              \item 抽气: 抽气的压强有两种代法
                    \begin{itemize}

                        \item 视为等温扩容
                              $$
                                  P_{1}V_{1}
                                  =
                                  P_{2}(V_{1}+\triangle V)
                              $$

                        \item 按比例抽走部分气体后,剩余气体重新充满原体积
                              $$
                                  P_{1}V_{1}
                                  -
                                  P_{1}\triangle V
                                  =
                                  P_{2}V_{1}
                              $$
                        \item $n$次抽气模型(等温),压强等比递减
                              $$
                                  P_nV - P_n\triangle V = P_{n+1}V
                              $$
                              $$
                                  P_{n+1}
                                  =
                                  P_n
                                  \left(
                                  1-\frac{\triangle V}{V}
                                  \right)
                              $$
                              $$
                                  P_n
                                  =
                                  P_0
                                  \left(
                                  1-\frac{\triangle V}{V}
                                  \right)^n
                              $$

                    \end{itemize}
          \end{enumerate}
\end{itemize}

\subsubsection{固体}
\begin{enumerate}

              \item 根据有无规则的外形\textbf{晶体}分为: 单晶体(有) \, 多晶体(无)

                        \hspace{3.8em}其晶体结构是连续的,或者可以说,在宏观尺度范围内单晶不包含晶界.

                        \,

                        多晶体: 仅存在于固体,由多颗大小及方向各异的晶粒所构成

                        \hspace{3.8em}而这些晶粒一般都由大量微小的单晶或微晶(微结晶、结晶粒、结晶子)组成.

              \item 常见晶体: 石英 \, 食盐 \, 明矾 \, 云母 \, 天然水晶
              \item 常见非晶体: 蜂蜡 \, 橡浆 \, 玻璃

    \item 各向同性与各向异性:
          \begin{itemize}
              \item 定义: 各个方向上的物理性质的同异(主要指导电性,导热性,透光性等)
              \item 单晶体: 其具有规则外形,因此不同方向上物理性质具有差异,各向异性
              \item 多晶体与非晶体: 其具有不规则外形,因此不同方向上物理性质一致,各向同性
          \end{itemize}

    \item 总结:

          \hspace{3em}\begin{tabular}{|c|c|c|c|}
              \hline
                   & 单晶体  & 多晶体  & 非晶体  \\
              \hline
              外形   & 规则   & 无规则  & 无规则  \\
              \hline
              固定熔点 & 有    & 有    & 无    \\
              \hline
              物理性质 & 各向异性 & 各向同性 & 各向同性 \\
              \hline
          \end{tabular}
\end{enumerate}

          \end{itemize}

          \begin{itemize}
              \item 温度: 温度越高,蒸发与冷凝强度均高,分子密度也越大,蒸汽的饱和汽压越大
              \item 体积: 与液面上方的体积\textbf{无关}(改变体积会改变当前状态,稳态不变)
              \item 密度: 无关。增大体积,上方蒸汽密度下降,液化能力下降,蒸发能力不变

                    \hspace{2.7em}上方净累积分子,稳态后,密度不变
          \end{itemize}
\end{enumerate}

              \item 表达式: $ \triangle u = W + Q $

                    \begin{center}
                        \begin{tabular}{|c|c|c|c|}
                            \hline
                            \,  & $W$      & $Q$  & $\triangle u$ \\
                            \hline
                            $+$ & 外界对物体做正功 & 吸收热量 & 内能增加          \\
                            \hline
                            $-$ & 物体对外界做正功 & 放出热量 & 内能减少          \\
                            \hline
                        \end{tabular}
                    \end{center}

              \item 气体膨胀/压缩时: 物体对外界做功/外界对物体做功
              \item 判断内能变化不同研究对象分析方式不同:
                    \begin{enumerate}[label = (\alph*)]
                        \item 研究对象为 \textbf{固体或液体} \, 时:

                              \begin{minipage}{0.8\textwidth}

                              \end{minipage}

                              $m$一定时,理想气体无分子势能,其内能只和温度有关
                    \end{enumerate}

              \item 经典电磁理论,电子辐射的电磁波的频率就是其绕核转动频率.电子越转能量越小,那么离原子核就越来越近,转的也就越来越快,

          \end{itemize}

\subsection{核子的概念}

\begin{enumerate}
    \item 原子核由两种基本粒子构成,它们统称为\textbf{核子}:

          \begin{itemize}
              \item 质子(proton):带正电 \quad 符号 $p$
              \item 中子(neutron):不带电 \quad 符号 $n$
          \end{itemize}

    \item 原子核中:
          \begin{itemize}
              \item 质子数为 $Z$
              \item 中子数为 $N$
          \end{itemize}
          则核子总数(质量数)为:
          $$
              A = Z + N
          $$

          原子核的表示: $_{Z}^{A}X$ \quad ($A$表质量数,$Z$为原子核的电荷数 \, eg. $_{2}^{4}He$)

          \vspace{-1em}
          \hspace{-0.9em}\begin{adjustbox}{minipage=0.305\linewidth, bgcolor=gray!20, padding=1em}
              \small
              同位素: \, 质子数一样,质量数不一样
          \end{adjustbox}
          \vspace{-1em}
\end{enumerate}

          \begin{itemize}
              \item $\alpha$衰变: 放射性元素原子核整体发射$\alpha$粒子

                    \hspace{3.6em}速度: 0.1c (穿透弱,电荷量大->电离能力强)
                    $$

                    $$
              \item $\beta$衰变:放射性元素原子核的中子转变为质子与电子(\textbf{非核外电子})

                        \end{aligned}
                    $$

                    $$

              \item $\gamma$衰变: $\alpha$衰变与$\beta$衰变过程中通常会伴随着$\gamma$射线(光,并没有释放$\nuc{1}{0}{n}$)
                    $$
                        \begin{aligned}
                            \text{(1) }\alpha\text{衰变伴随}\gamma:\quad

\subsection{质能方程}

\begin{itemize}
    \item 在核反应方程中,\textbf{质量数与质子数守恒},但\textbf{质量并不守恒}
    \item 质子和中子的质量数均为 $1$,但实际质量略有差异
    \item 质能方程:
          $$
              E = mc^2 \quad (m:\text{kg},\ E:\text{J})
          $$
    \item 质量亏损对应能量:
          $$
              \Delta E = \Delta m \, c^2
          $$
    \item 质量单位: \, $1u$ \, (原子质量单位,定义为碳原子质量的 $\frac{1}{12}$)
    \item 能量单位(默认乘过$c^{2}$):
          $$
              1u \approx 931.5\text{ MeV}, \quad
              1\text{ MeV} = 1.6 \times 10^{-13}\text{ J}
          $$
\end{itemize}

\subsection{结合能与比结合能}

\begin{enumerate}
    \item 原子核的核子之间存在强相互作用:\textbf{核力}
          \begin{enumerate}[label = (\alph*)]
              \item 在原子核尺度内,核力远大于库仑力
              \item 核力为短程力,作用范围约为 $1.5\times10^{-15}\text{ m}$

                    当核子间距离处于该量级时表现为吸引力,距离过小时表现为排斥作用

                    \vspace{-1em}\hspace{-0.6em}\begin{adjustbox}{minipage=0.545\linewidth, bgcolor=gray!20, padding=0.8em}
                        \small
                        自然界中较轻原子核,质子数与中子数大致相等;

                        对于较重原子核,中子数大于质子数,且原子序数越大差值越大
                    \end{adjustbox}
          \end{enumerate}

    \item 结合能

          \begin{itemize}
              \item 定义: 把原子核中的核子完全分开所需要的能量称为结合能$E_{bind}$
              \item 原子核的质量不能简单看作质子和中子质量的相加,因为实际存在质量亏损
              \item 原子核的质量\textbf{小于}组成它的核子质量之和,
                    其质量的差值对应能量为\textbf{结合能}
          \end{itemize}

    \item 比结合能

          比结合能$E_{0}$: 一般的,核子数越多,总结合能越大,因此更有意义的是比较其平均值

          设原子核结合能为 $E_{bind}$,核子数为 $A$,则比结合能$E_{0}$:
          $$
              E_0 = \frac{E_{bind}}{A}
          $$

          比结合能越大,表示核子结合越牢,原子核越稳定
\end{enumerate}

\subsection{具体核反应的概念辨析}

以铀核裂变反应为例:

\[
    \nuc{235}{92}{U} + \nuc{1}{0}{n} \rightarrow \nuc{141}{56}{Ba} + \nuc{92}{36}{Kr} + 3\nuc{1}{0}{n} + \text{能量}
\]

\begin{enumerate}[label = (\arabic*)]

    \item \textbf{质量亏损与放能}

          设反应前总质量为 $m_1$,反应后总质量为 $m_2$

          若 $m_1 > m_2$,则存在质量亏损:
          $$
              \Delta m = m_1 - m_2
          $$

          对应释放能量:
          $$
              E = \Delta m \, c^2
          $$

          说明该反应为放能反应

    \item \textbf{质量亏损的本质——结合能变化}

          该反应中,生成物的总质量小于反应物的总质量,

          说明发生了质量亏损,对应释放能量：
          $$
              \triangle E = \Delta m \, c^2
          $$
    \item \textbf{从结合能角度判断反应}

          比较反应前后总结合能:

          放能形成了\textbf{更加稳定的生成物},其\textbf{总能量更低}

          因此\textbf{生成物总结合能}$>$ \textbf{反应物总结合能}

          多出的结合能以能量形式释放

    \item \textbf{从比结合能角度理解稳定性}

          裂变过程中: 重核(铀) $\rightarrow$ 中等质量核(钡、氪)

          后者原子核更稳定,核子结合得牢固,比结合能增大

    \item \textbf{统一理解}

          核反应的本质: \textbf{反应方向由原子核稳定性决定}

          \hspace{6.7em}\textbf{通常向着原子核更稳定的方向发生(即能量更低的状态)}

          \hspace{6.7em}但是否发生还取决于具体条件

          $$
              \text{能量降低} \llra \text{放能(质量亏损)} \llra \text{稳定性增} \llra \text{总结合能增} \llra \text{比结合能增}
          $$
\end{enumerate}
